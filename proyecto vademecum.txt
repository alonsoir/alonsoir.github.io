Podría procesar el csv de tal manera que con un map clave valor, donde la clave es el nombre de la enfermedad y el valor es una lista o un conjunto de los sintomas.

  def loadVademecum() : Map[String, Set<String>] = {
    
    // Create a Map of Ints to Strings, and populate it from u.item.
    var vademecum:Map[String, Set<String>] = Map()
    
     val lines = Source.fromFile("PATH_TO_VADEMECUM").getLines()
     var symptons:Set<String> = Set()

     for (line <- lines) {
       var fields = line.split(',')
       for (index <- 1 to fields.length){
       		symptons += fields(index)
       }
       if (fields.length > 1) {
        vademecum += (fields(0).toInt -> symptons)
       }
     }
    
     return vademecum
  }
